  ],
  arrears: [
    { no: 2, name: "Subject 2", attempt: "2nd Attempt", total: 70, result: "PASS" },

const form = document.getElementById("search-form");
const resultSection = document.getElementById("result-section");
const errorMsg = document.getElementById("error-msg");

form.addEventListener("submit", e => {
  e.preventDefault();
  const entered = document.getElementById("regno").value.trim();
  if (entered === studentData.regno) {
    displayResult();
    errorMsg.textContent = "";
  } else {
    resultSection.style.display = "none";
    errorMsg.textContent = "Register Number not found. Please check and try again.";
  }
});

function displayResult() {
  document.getElementById("r-reg").textContent = studentData.regno;
  document.getElementById("r-name").textContent = studentData.name;
  document.getElementById("r-dob").textContent = studentData.dob;

  const main = document.getElementById("main-subjects");
  main.innerHTML = "";
  studentData.mainSubjects.forEach(s => {
    main.innerHTML += `<tr><td>${s.no}</td><td>${s.name}</td><td>${s.internal}</td>
      <td>${s.external}</td><td>${s.total}</td><td>${s.grade}</td><td>${s.result}</td></tr>`;
  });

  const arr = document.getElementById("arrear-subjects");
  arr.innerHTML = "";
  studentData.arrears.forEach(s => {
    arr.innerHTML += `<tr><td>${s.no}</td><td>${s.name}</td><td>${s.attempt}</td>
      <td>${s.total}</td><td>${s.result}</td></tr>`;
  });

  resultSection.style.display = "block";
}
